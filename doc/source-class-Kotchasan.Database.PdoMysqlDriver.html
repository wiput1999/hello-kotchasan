193:         } else {

206:             } else {

211:                     }
212:                 } elseif (isset($sqls['update'])) {
213:                     $sql .= 'UPDATE '.$sqls['update'];
214:                 } elseif (isset($sqls['delete'])) {
215:                     $sql .= 'DELETE FROM '.$sqls['delete'];
216:                 }
217:             }

222:             }
